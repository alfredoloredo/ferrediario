<html lang="es">

<head>

    <meta charset="UTF-8">

    <meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1.0, user-scalable=no">

    <title>Feedback Diario</title>

    <!-- Cargamos todo de forma ultra-segura -->

    <script src="https://cdn.tailwindcss.com"></script>

    <script src="https://unpkg.com/react@18/umd/react.production.min.js"></script>

    <script src="https://unpkg.com/react-dom@18/umd/react-dom.production.min.js"></script>

    <script src="https://unpkg.com/@babel/standalone/babel.min.js"></script>

    <!-- Iconos Lucide (Versión estable) -->

    <script src="https://unpkg.com/lucide@0.344.0/dist/umd/lucide.min.js"></script>

    <style>

        @keyframes fall { to { transform: translateY(100vh) rotate(360deg); opacity: 0; } }

        .confetti { position: fixed; z-index: 100; pointer-events: none; animation: fall 3s linear forwards; }

        .no-scrollbar::-webkit-scrollbar { display: none; }

        .bg-ai { background: linear-gradient(135deg, #6366f1 0%, #a855f7 100%); }

    </style>

</head>

<body class="bg-slate-50 font-sans">

    <div id="root"></div>


    <script type="text/babel">

        const { useState, useEffect, useMemo } = React;


        function App() {

            // --- ESTADOS ---

            const [view, setView] = useState('home');

            const [formData, setFormData] = useState({ empleado: '', sucursal: 'Tancol' });

            const [activeCategory, setActiveCategory] = useState(null);

            const [records, setRecords] = useState(() => {

                const saved = localStorage.getItem('ferre_logs');

                return saved ? JSON.parse(saved) : [];

            });

            const [confetti, setConfetti] = useState([]);


            // Datos fijos

            const empleados = ['Adrian', 'Ana', 'Emi', 'Fred', 'Sury'];

            const sucursales = ['Tancol', 'Insurgentes'];

            const categorias = [

                { id: 'perdida', t: 'Venta Perdida', c: 'bg-red-500', i: 'x-circle' },

                { id: 'devol', t: 'Devolución', c: 'bg-orange-500', i: 'rotate-ccw' },

                { id: 'cliente', t: 'Cliente', c: 'bg-blue-500', i: 'message-square' },

                { id: 'stock', t: 'Stock', c: 'bg-emerald-500', i: 'package' },

                { id: 'insumos', t: 'Insumos', c: 'bg-purple-500', i: 'package-open' }

            ];


            // Guardar en memoria local automáticamente

            useEffect(() => {

                localStorage.setItem('ferre_logs', JSON.stringify(records));

            }, [records]);


            // Forzar iconos de Lucide

            useEffect(() => {

                if (window.lucide) window.lucide.createIcons();

            });


            // --- FUNCIONES ---

            const handleSave = () => {

                const newLog = {

                    ...formData,

                    id: Date.now(),

                    cat: activeCategory.t,

                    color: activeCategory.c,

                    time: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })

                };

                setRecords([newLog, ...records]);

                

                // Efecto confeti

                setConfetti(Array.from({ length: 20 }).map((_, i) => ({ id: i, x: Math.random() * 100 })));

                setView('success');

                

                setTimeout(() => {

                    setView('home');

                    setConfetti([]);

                }, 3000);

            };


            // --- VISTAS ---

            if (view === 'success') {

                return (

                    <div className="flex flex-col items-center justify-center min-h-screen bg-white p-10 text-center">

                        <div className="w-24 h-24 bg-emerald-500 rounded-[2rem] flex items-center justify-center mb-6 shadow-xl shadow-emerald-100">

                            <i data-lucide="check" className="text-white w-12 h-12"></i>

                        </div>

                        <h1 className="text-4xl font-black italic uppercase text-slate-900">¡LISTO!</h1>

                        <p className="text-slate-400 font-bold mt-2 uppercase tracking-widest text-xs">Registro Guardado</p>

                    </div>

                );

            }


            return (

                <div className="max-w-md mx-auto min-h-screen flex flex-col relative">

                    {confetti.map(p => <div key={p.id} className="confetti w-3 h-3 bg-yellow-400 rounded-full" style={{left: p.x + '%'}} />)}

                    

                    <header className="bg-slate-900 p-5 flex justify-between items-center text-white sticky top-0 z-50">

                        <div className="flex items-center gap-3">

                            <div className="w-10 h-10 bg-yellow-400 rounded-xl flex items-center justify-center text-slate-900 font-black text-xl shadow-lg">F</div>

                            <h1 className="font-black italic text-xl tracking-tighter uppercase">Feedback Diario</h1>

                        </div>

                        <button onClick={() => setView(view === 'dash' ? 'home' : 'dash')} className="p-2 bg-slate-800 rounded-xl">

                            <i data-lucide={view === 'dash' ? 'home' : 'bar-chart-3'} className="text-yellow-400"></i>

                        </button>

                    </header>


                    <main className="p-4 space-y-6 flex-1">

                        {view === 'home' && (

                            <div className="space-y-6">

                                {/* Selector Equipo */}

                                <div className="bg-white p-5 rounded-[2.5rem] shadow-sm border border-slate-100 space-y-5">

                                    <div className="flex gap-2">

                                        {sucursales.map(s => (

                                            <button key={s} onClick={() => setFormData({...formData, sucursal: s})} className={`flex-1 py-3 rounded-xl text-[11px] font-black uppercase border-2 transition-all ${formData.sucursal === s ? 'bg-blue-600 text-white border-blue-600' : 'bg-slate-50 text-slate-400 border-transparent'}`}>{s}</button>

                                        ))}

                                    </div>

                                    <div className="grid grid-cols-5 gap-1.5">

                                        {empleados.map(e => (

                                            <button key={e} onClick={() => setFormData({...formData, empleado: e})} className={`py-5 rounded-xl text-[10px] font-black border-2 transition-all ${formData.empleado === e ? 'bg-slate-800 text-white border-slate-800 scale-105 shadow-lg' : 'bg-slate-50 text-slate-400 border-transparent'}`}>{e}</button>

                                        ))}

                                    </div>

                                </div>


                                {/* Botones de Registro */}

                                <div className="grid grid-cols-5 gap-2">

                                    {categorias.map(cat => (

                                        <button key={cat.id} onClick={() => { if(!formData.empleado) return alert('Selecciona tu nombre'); setActiveCategory(cat); setView('form'); }} className="flex flex-col items-center gap-2 active:scale-95 transition-transform">

                                            <div className={`${cat.c} text-white p-4 rounded-3xl shadow-lg w-full aspect-square flex items-center justify-center`}>

                                                <i data-lucide={cat.i}></i>

                                            </div>

                                            <span className="text-[9px] font-black uppercase text-slate-500">{cat.t}</span>

                                        </button>

                                    ))}

                                </div>


                                {/* Lista Reciente */}

                                <div className="space-y-3">

                                    <h3 className="text-[10px] font-black uppercase text-slate-400 tracking-widest px-1">Últimos movimientos</h3>

                                    {records.slice(0, 3).map(r => (

                                        <div key={r.id} className="bg-white p-4 rounded-2xl border border-slate-100 flex items-center gap-4 shadow-sm">

                                            <div className={`${r.color} w-2 h-10 rounded-full`}></div>

                                            <div className="flex-1">

                                                <p className="font-black text-slate-800 text-sm uppercase">{r.cat}</p>

                                                <p className="text-[10px] font-bold text-slate-400 uppercase">{r.empleado} • {r.sucursal} • {r.time}</p>

                                            </div>

                                        </div>

                                    ))}

                                </div>

                            </div>

                        )}


                        {view === 'form' && (

                            <div className="bg-white rounded-[2.5rem] shadow-2xl overflow-hidden border border-slate-100 animate__animated animate__slideInUp">

                                <div className="p-8 border-b flex justify-between items-center">

                                    <h2 className="font-black text-2xl uppercase italic text-slate-800">{activeCategory.t}</h2>

                                    <button onClick={() => setView('home')} className="text-slate-300"><i data-lucide="x"></i></button>

                                </div>

                                <div className="p-8 space-y-6">

                                    <div className="space-y-4">

                                        <label className="text-[10px] font-black uppercase text-slate-400 tracking-widest">Información del Reporte</label>

                                        <input type="text" placeholder="¿Qué sucedió? (Producto, motivo, etc)" className="w-full p-4 bg-slate-50 border-2 border-slate-100 rounded-2xl font-bold outline-none focus:border-blue-500 transition-all" onChange={e => setFormData({...formData, info: e.target.value})} />

                                        <textarea placeholder="Notas adicionales..." className="w-full p-4 bg-slate-50 border-2 border-slate-100 rounded-2xl font-medium text-sm h-32 outline-none" onChange={e => setFormData({...formData, notas: e.target.value})}></textarea>

                                    </div>

                                    <button onClick={handleSave} className={`${activeCategory.c} w-full text-white p-5 rounded-2xl font-black text-xl uppercase shadow-xl active:scale-95 transition-all`}>Guardar Reporte</button>

                                </div>

                            </div>

                        )}


                        {view === 'dash' && (

                            <div className="space-y-6">

                                <div className="bg-ai p-6 rounded-[2.5rem] text-white shadow-xl">

                                    <div className="flex items-center gap-2 mb-2">

                                        <i data-lucide="brain"></i>

                                        <h3 className="font-black uppercase text-xs">Análisis Inteligente</h3>

                                    </div>

                                    <p className="text-[11px] font-medium opacity-80">La IA estará lista cuando conectes tu llave de Gemini. Por ahora, aquí verás tus métricas locales.</p>

                                </div>

                                <div className="bg-white p-6 rounded-[2.5rem] border border-slate-100 shadow-sm">

                                    <h3 className="font-black uppercase text-xs text-slate-400 mb-4 tracking-widest">Registros Totales: {records.length}</h3>

                                    <div className="space-y-4">

                                        {records.map(r => (

                                            <div key={r.id} className="text-[11px] font-bold border-b border-slate-50 pb-2 flex justify-between uppercase">

                                                <span className="text-slate-800">{r.cat}</span>

                                                <span className="text-slate-300">{r.time}</span>

                                            </div>

                                        ))}

                                    </div>

                                </div>

                            </div>

                        )}

                    </main>

                </div>

            );

        }


        const root = ReactDOM.createRoot(document.getElementById('root'));

        root.render(<App />);

    </script>

</body>

</html>
